11/// DuckDB connector for local analytics
12pub struct DuckDbConnector {
13    pool: sqlx::SqlitePool,
14    _database_path: String,
15}

17impl DuckDbConnector {
18    /// Create a new DuckDB connector
19    ///
20    /// Note: We use SQLite as a stand-in since sqlx doesn't support DuckDB directly.
21    /// For full DuckDB support, we'd use the `duckdb` crate.
22    /// This implementation provides similar functionality for local file queries.
23    pub async fn new(database_path: Option<&str>) -> Result<Self> {
24        let path = database_path.unwrap_or(":memory:");
25
26        let pool = sqlx::SqlitePool::connect(&format!("sqlite:{}", path))
27            .await
28            .context("Failed to create local analytics database")?;
29
30        Ok(Self {
31            pool,
32            _database_path: path.to_string(),
33        })
34    }

36    /// Execute a SQL query
37    pub async fn execute_sql(
38        &self,
39        query: &str,
40        limit: Option<usize>,
41    ) -> Result<(Vec<String>, Vec<HashMap<String, String>>)> {
42        // Add LIMIT if specified
43        let query = if let Some(limit) = limit {
44            if !query.to_lowercase().contains("limit") {
45                format!("{} LIMIT {}", query.trim_end_matches(';'), limit)
46            } else {
47                query.to_string()
48            }
49        } else {
50            query.to_string()
51        };
52
53        let rows: Vec<sqlx::sqlite::SqliteRow> = sqlx::query(&query)
54            .fetch_all(&self.pool)
55            .await
56            .context("Query execution failed")?;
57
58        if rows.is_empty() {
59            return Ok((vec![], vec![]));
60        }
61
62        use sqlx::Row;
63        let columns: Vec<String> = rows[0]
64            .columns()
65            .iter()
66            .map(|c| c.name().to_string())
67            .collect();
68
69        let result_rows: Vec<HashMap<String, String>> = rows
70            .iter()
71            .map(|row| {
72                columns
73                    .iter()
74                    .enumerate()
75                    .map(|(i, col)| {
76                        let value: String = row
77                            .try_get::<String, _>(i)
78                            .or_else(|_| row.try_get::<i64, _>(i).map(|v| v.to_string()))
79                            .or_else(|_| row.try_get::<f64, _>(i).map(|v| v.to_string()))
80                            .or_else(|_| row.try_get::<bool, _>(i).map(|v| v.to_string()))
81                            .unwrap_or_else(|_| "NULL".to_string());
82                        (col.clone(), value)
83                    })
84                    .collect()
85            })
86            .collect();
87
88        Ok((columns, result_rows))
89    }

91    /// Query a CSV file directly
92    pub async fn query_csv(&self, file_path: &str, query: Option<&str>) -> Result<(Vec<String>, Vec<HashMap<String, String>>)> {
93        // Read CSV and create temporary table
94        let path = Path::new(file_path);
95        if !path.exists() {
96            anyhow::bail!("File not found: {}", file_path);
97        }
98
99        // For SQLite, we need to import the CSV first
100        // In a real DuckDB implementation, we'd use: SELECT * FROM read_csv_auto('file.csv')
101
102        let table_name = path
103            .file_stem()
104            .and_then(|s| s.to_str())
105            .unwrap_or("csv_data")
106            .replace(|c: char| !c.is_alphanumeric(), "_");
107
108        // Read CSV file
109        let content = std::fs::read_to_string(file_path)
110            .context("Failed to read CSV file")?;
111
112        let mut lines = content.lines();
113        let header = lines.next().context("Empty CSV file")?;
114        let columns: Vec<&str> = header.split(',').map(|s| s.trim()).collect();
115
116        // Create table
117        let column_defs: Vec<String> = columns
118            .iter()
119            .map(|c| format!("\"{}\" TEXT", c.replace('"', "")))
120            .collect();
121
122        let create_sql = format!(
123            "CREATE TEMP TABLE IF NOT EXISTS {} ({})",
124            table_name,
125            column_defs.join(", ")
126        );
127
128        sqlx::query(&create_sql)
129            .execute(&self.pool)
130            .await
131            .context("Failed to create temp table")?;
132
133        // Insert data
134        for line in lines {
135            let values: Vec<&str> = line.split(',').map(|s| s.trim()).collect();
136            if values.len() == columns.len() {
137                let placeholders: Vec<String> = (1..=values.len()).map(|i| format!("?{}", i)).collect();
138                let insert_sql = format!(
139                    "INSERT INTO {} VALUES ({})",
140                    table_name,
141                    placeholders.join(", ")
142                );
143
144                let mut query_builder = sqlx::query(&insert_sql);
145                for value in &values {
146                    query_builder = query_builder.bind(*value);
147                }
148
149                let _ = query_builder.execute(&self.pool).await;
150            }
151        }
152
153        // Execute query
154        let default_query = format!("SELECT * FROM {}", table_name);
155        let sql = query.unwrap_or(&default_query);
156        self.execute_sql(sql, None).await
157    }

159    /// Query a JSON file
160    pub async fn query_json(&self, file_path: &str) -> Result<(Vec<String>, Vec<HashMap<String, String>>)> {
161        let content = std::fs::read_to_string(file_path)
162            .context("Failed to read JSON file")?;
163
164        let data: serde_json::Value = serde_json::from_str(&content)
165            .context("Failed to parse JSON")?;
166
167        match data {
168            serde_json::Value::Array(arr) => {
169                if arr.is_empty() {
170                    return Ok((vec![], vec![]));
171                }
172
173                // Get columns from first object
174                let columns: Vec<String> = if let Some(serde_json::Value::Object(obj)) = arr.first() {
175                    obj.keys().cloned().collect()
176                } else {
177                    vec!["value".to_string()]
178                };
179
180                let rows: Vec<HashMap<String, String>> = arr
181                    .into_iter()
182                    .filter_map(|v| {
183                        if let serde_json::Value::Object(obj) = v {
184                            Some(
185                                obj.into_iter()
186                                    .map(|(k, v)| {
187                                        let value = match v {
188                                            serde_json::Value::String(s) => s,
189                                            serde_json::Value::Number(n) => n.to_string(),
190                                            serde_json::Value::Bool(b) => b.to_string(),
191                                            serde_json::Value::Null => "NULL".to_string(),
192                                            _ => v.to_string(),
193                                        };
194                                        (k, value)
195                                    })
196                                    .collect()
197                            )
198                        } else {
199                            None
200                        }
201                    })
202                    .collect();
203
204                Ok((columns, rows))
205            }
206            serde_json::Value::Object(obj) => {
207                let columns: Vec<String> = obj.keys().cloned().collect();
208                let row: HashMap<String, String> = obj
209                    .into_iter()
210                    .map(|(k, v)| {
211                        let value = match v {
212                            serde_json::Value::String(s) => s,
213                            serde_json::Value::Number(n) => n.to_string(),
214                            serde_json::Value::Bool(b) => b.to_string(),
215                            serde_json::Value::Null => "NULL".to_string(),
216                            _ => v.to_string(),
217                        };
218                        (k, value)
219                    })
220                    .collect();
221                Ok((columns, vec![row]))
222            }
223            _ => anyhow::bail!("JSON must be an array or object"),
224        }
225    }

227    /// List tables in the database
228    pub async fn list_tables(&self) -> Result<Vec<String>> {
229        let (_, rows) = self
230            .execute_sql(
231                "SELECT name FROM sqlite_master WHERE type='table' ORDER BY name",
232                None,
233            )
234            .await?;
235        Ok(rows
236            .into_iter()
237            .filter_map(|r| r.get("name").cloned())
238            .collect())
239    }

241    /// Analyze a file and return statistics
242    pub async fn analyze_file(&self, file_path: &str) -> Result<HashMap<String, String>> {
243        let path = Path::new(file_path);
244        let extension = path.extension().and_then(|s| s.to_str()).unwrap_or("");
245
246        let mut stats = HashMap::new();
247        stats.insert("file".to_string(), file_path.to_string());
248        stats.insert("extension".to_string(), extension.to_string());
249
250        let metadata = std::fs::metadata(file_path)?;
251        stats.insert("size".to_string(), humansize::format_size(metadata.len(), humansize::BINARY));
252
253        match extension.to_lowercase().as_str() {
254            "csv" => {
255                let (columns, rows) = self.query_csv(file_path, Some("SELECT COUNT(*) as count FROM csv_data")).await?;
256                if let Some(row) = rows.first() {
257                    if let Some(count) = row.get("count") {
258                        stats.insert("rows".to_string(), count.clone());
259                    }
260                }
261                stats.insert("columns".to_string(), columns.len().to_string());
262            }
263            "json" => {
264                let (columns, rows) = self.query_json(file_path).await?;
265                stats.insert("rows".to_string(), rows.len().to_string());
266                stats.insert("columns".to_string(), columns.len().to_string());
267            }
268            _ => {}
269        }
270
271        Ok(stats)
272    }
