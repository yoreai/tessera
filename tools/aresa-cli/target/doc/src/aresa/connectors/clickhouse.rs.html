1//! ClickHouse connector

3//! Connect to ClickHouse databases via HTTP interface.

5use anyhow::{Context, Result};
6use serde::Deserialize;
7use std::collections::HashMap;
8
9/// ClickHouse connector using HTTP interface
10pub struct ClickHouseConnector {
11    base_url: String,
12    client: reqwest::Client,
13    database: Option<String>,
14    username: Option<String>,
15    password: Option<String>,
16}
17
18#[derive(Deserialize)]
19struct ClickHouseResponse {
20    data: Vec<HashMap<String, serde_json::Value>>,
21    meta: Vec<ColumnMeta>,
22    rows: usize,
23    statistics: Option<Statistics>,
24}
25
26#[derive(Deserialize)]
27struct ColumnMeta {
28    name: String,
29    #[serde(rename = "type")]
30    _column_type: String,
31}
32
33#[derive(Deserialize)]
34struct Statistics {
35    elapsed: f64,
36    rows_read: u64,
37    bytes_read: u64,
38}
39
40impl ClickHouseConnector {
41    /// Create a new ClickHouse connector
42    pub async fn new(
43        host: &str,
44        port: Option<u16>,
45        database: Option<&str>,
46        username: Option<&str>,
47        password: Option<&str>,
48    ) -> Result<Self> {
49        let port = port.unwrap_or(8123);
50        let base_url = format!("http://{}:{}", host, port);
51
52        let connector = Self {
53            base_url,
54            client: reqwest::Client::new(),
55            database: database.map(String::from),
56            username: username.map(String::from),
57            password: password.map(String::from),
58        };
59
60        // Test connection
61        connector.test_connection().await?;
62
63        Ok(connector)
64    }
65
66    /// Execute a SQL query
67    pub async fn execute_sql(
68        &self,
69        query: &str,
70        limit: Option<usize>,
71    ) -> Result<(Vec<String>, Vec<HashMap<String, String>>)> {
72        // Add LIMIT if specified
73        let query = if let Some(limit) = limit {
74            if !query.to_lowercase().contains("limit") {
75                format!("{} LIMIT {}", query.trim_end_matches(';'), limit)
76            } else {
77                query.to_string()
78            }
79        } else {
80            query.to_string()
81        };
82
83        // Build request
84        let mut request = self.client.post(&self.base_url);
85
86        // Add authentication if provided
87        if let (Some(user), Some(pass)) = (&self.username, &self.password) {
88            request = request.basic_auth(user, Some(pass));
89        }
90
91        // Add database parameter
92        if let Some(db) = &self.database {
93            request = request.query(&[("database", db)]);
94        }
95
96        // Request JSON format
97        let query_with_format = format!("{} FORMAT JSON", query);
98
99        let response = request
100            .body(query_with_format)
101            .send()
102            .await
103            .context("Failed to send request to ClickHouse")?;
104
105        if !response.status().is_success() {
106            let error = response.text().await.unwrap_or_default();
107            anyhow::bail!("ClickHouse error: {}", error);
108        }
109
110        let result: ClickHouseResponse = response
111            .json()
112            .await
113            .context("Failed to parse ClickHouse response")?;
114
115        // Extract columns
116        let columns: Vec<String> = result.meta.iter().map(|m| m.name.clone()).collect();
117
118        // Convert data to string map
119        let rows: Vec<HashMap<String, String>> = result
120            .data
121            .into_iter()
122            .map(|row| {
123                row.into_iter()
124                    .map(|(k, v)| {
125                        let value = match v {
126                            serde_json::Value::String(s) => s,
127                            serde_json::Value::Number(n) => n.to_string(),
128                            serde_json::Value::Bool(b) => b.to_string(),
129                            serde_json::Value::Null => "NULL".to_string(),
130                            _ => v.to_string(),
131                        };
132                        (k, value)
133                    })
134                    .collect()
135            })
136            .collect();
137
138        Ok((columns, rows))
139    }

141    /// Test connection
142    pub async fn test_connection(&self) -> Result<()> {
143        let mut request = self.client.get(format!("{}/ping", self.base_url));
144
145        if let (Some(user), Some(pass)) = (&self.username, &self.password) {
146            request = request.basic_auth(user, Some(pass));
147        }
148
149        let response = request.send().await.context("Failed to ping ClickHouse")?;
150
151        if !response.status().is_success() {
152            anyhow::bail!("ClickHouse ping failed");
153        }
154
155        Ok(())
156    }

158    /// List databases
159    pub async fn list_databases(&self) -> Result<Vec<String>> {
160        let (_, rows) = self.execute_sql("SHOW DATABASES", None).await?;
161        Ok(rows
162            .into_iter()
163            .filter_map(|r| r.get("name").cloned())
164            .collect())
165    }
166
167    /// List tables in current database
168    pub async fn list_tables(&self) -> Result<Vec<String>> {
169        let (_, rows) = self.execute_sql("SHOW TABLES", None).await?;
170        Ok(rows
171            .into_iter()
172            .filter_map(|r| r.get("name").cloned())
173            .collect())
174    }
175
176    /// Get table schema
177    pub async fn describe_table(&self, table: &str) -> Result<Vec<HashMap<String, String>>> {
178        let (_, rows) = self
179            .execute_sql(&format!("DESCRIBE TABLE {}", table), None)
180            .await?;
181        Ok(rows)
182    }
183
184    /// Get system information
185    pub async fn system_info(&self) -> Result<HashMap<String, String>> {
186        let (_, rows) = self
187            .execute_sql(
188                "SELECT version() as version, uptime() as uptime_seconds",
189                None,
190            )
191            .await?;
192        Ok(rows.into_iter().next().unwrap_or_default())
193    }
