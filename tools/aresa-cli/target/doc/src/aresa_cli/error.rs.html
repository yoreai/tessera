7/// Main error type for ARESA CLI
8#[derive(Error, Debug)]
9pub enum AresaError {
10    /// Configuration errors
11    #[error("Configuration error: {0}")]
12    Config(String),
13
14    /// Connection errors
15    #[error("Connection failed: {0}")]
16    Connection(#[from] ConnectionError),
17
18    /// Query errors
19    #[error("Query error: {0}")]
20    Query(String),
21
22    /// Authentication errors
23    #[error("Authentication failed: {0}")]
24    Auth(String),
25
26    /// Timeout errors
27    #[error("Operation timed out after {0}s")]
28    Timeout(u64),
29
30    /// File system errors
31    #[error("Filesystem error: {0}")]
32    Filesystem(String),
33
34    /// LLM/NLP errors
35    #[error("NLP error: {0}")]
36    Nlp(String),
37
38    /// Validation errors
39    #[error("Validation error: {0}")]
40    Validation(String),
41
42    /// Generic errors
43    #[error("{0}")]
44    Other(#[from] anyhow::Error),
45}

81/// Result type alias for ARESA operations
82pub type Result<T> = std::result::Result<T, AresaError>;
83
84/// Helper trait for adding context to errors
85pub trait ResultExt<T> {
86    /// Add context to an error
87    fn with_context<F, S>(self, f: F) -> Result<T>
88    where
89        F: FnOnce() -> S,
90        S: Into<String>;
91}
92
93impl<T, E: std::error::Error + Send + Sync + 'static> ResultExt<T> for std::result::Result<T, E> {
94    fn with_context<F, S>(self, f: F) -> Result<T>
95    where
96        F: FnOnce() -> S,
97        S: Into<String>,
98    {
99        self.map_err(|e| AresaError::Other(anyhow::anyhow!("{}: {}", f().into(), e)))
100    }
101}

103/// Validate a connection string
104pub fn validate_connection_string(conn_str: &str, source_type: &str) -> std::result::Result<(), String> {
105    if conn_str.is_empty() {
106        return Err("Connection string cannot be empty".to_string());
107    }
108
109    match source_type.to_lowercase().as_str() {
110        "postgres" => {
111            if !conn_str.starts_with("postgres://") && !conn_str.starts_with("postgresql://") {
112                return Err("PostgreSQL connection string must start with 'postgres://' or 'postgresql://'".to_string());
113            }
114        }
115        "mysql" => {
116            if !conn_str.starts_with("mysql://") {
117                return Err("MySQL connection string must start with 'mysql://'".to_string());
118            }
119        }
120        "sqlite" => {
121            // SQLite can be a file path or :memory:
122            if conn_str != ":memory:" && !conn_str.ends_with(".db") && !conn_str.ends_with(".sqlite") {
123                // Just a warning, not an error
124            }
125        }
126        _ => {}
127    }
128
129    Ok(())
130}

132/// Validate a bucket name
133pub fn validate_bucket_name(bucket: &str) -> std::result::Result<(), String> {
134    if bucket.is_empty() {
135        return Err("Bucket name cannot be empty".to_string());
136    }
137
138    if bucket.len() < 3 || bucket.len() > 63 {
139        return Err("Bucket name must be between 3 and 63 characters".to_string());
140    }
141
142    if !bucket.chars().all(|c| c.is_ascii_lowercase() || c.is_ascii_digit() || c == '-' || c == '.') {
143        return Err("Bucket name can only contain lowercase letters, numbers, hyphens, and periods".to_string());
144    }
145
146    Ok(())
147}

149/// Sanitize SQL input (basic protection)
150pub fn sanitize_identifier(identifier: &str) -> String {
151    // Remove or escape potentially dangerous characters
152    identifier
153        .chars()
154        .filter(|c| c.is_alphanumeric() || *c == '_')
155        .collect()
156}
